// Создание нового контрагента с автодополнением типов
НовыйКонтрагент = Справочники.Контрагенты.СоздатьЭлемент();
// TypeRef: context:objects/CatalogObject.Контрагенты ✓
// Facet: Object ✓
// Доступные методы: Записать(), Удалить(), ЗаполнитьПоУмолчанию() ✓

НовыйКонтрагент.ИНН = "1234567890";
// Property type: Строка(12) ✓
// Access: Чтение/Запись ✓

// Автоматически генерируется runtime контракт:

ВызватьИсключение "Type mismatch: expected String or CatalogRef.Контрагенты";
КонецЕсли;
